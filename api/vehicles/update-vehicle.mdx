---
title: "Update Vehicle"
api: "PUT /api/vehicles/{id}"
description: "Update an existing vehicle"
---

Updates all fields of an existing vehicle. Returns 404 if the vehicle is not found.

## Path Parameters

<ParamField path="id" type="long" required>
  The unique identifier of the vehicle to update
</ParamField>

## Request Body

<ParamField body="reference" type="string" optional>
  Unique reference code
</ParamField>

<ParamField body="brand" type="string" required>
  Vehicle brand name
</ParamField>

<ParamField body="model" type="string" required>
  Vehicle model name
</ParamField>

<ParamField body="hp" type="integer" optional>
  Horsepower
</ParamField>

<ParamField body="autonomy" type="integer" optional>
  Autonomy/range in kilometers
</ParamField>

<ParamField body="averageconsumption" type="double" optional>
  Average fuel consumption
</ParamField>

<ParamField body="description" type="string" optional>
  Vehicle description
</ParamField>

<ParamField body="price" type="decimal" optional>
  Vehicle price
</ParamField>

<ParamField body="year" type="integer" optional>
  Manufacturing year
</ParamField>

<ParamField body="fuelType" type="string" optional>
  Fuel type (e.g., gasoline, diesel, electric, hybrid)
</ParamField>

<ParamField body="mileage" type="long" optional>
  Mileage in kilometers
</ParamField>

<ParamField body="extras" type="string" optional>
  Additional features and extras
</ParamField>

<ParamField body="doors" type="integer" optional>
  Number of doors
</ParamField>

<ParamField body="vehicleType" type="string" optional>
  Type of vehicle (e.g., sedan, suv, truck, motorcycle)
</ParamField>

<ParamField body="offers" type="string" optional>
  Special offers status (e.g., "Yes" or "No")
</ParamField>

<ParamField body="images" type="array" optional>
  Array of vehicle images
</ParamField>
