---
title: "Get Vehicle"
api: "GET /api/vehicles/{id}"
description: "Retrieve a specific vehicle by ID"
---

Returns details of a single vehicle identified by its unique ID.

## Path Parameters

<ParamField path="id" type="long" required>
  The unique identifier of the vehicle
</ParamField>

## Response

Returns a single Vehicle object or 404 if not found.

<ResponseField name="id" type="long">
  Unique identifier for the vehicle
</ResponseField>

<ResponseField name="reference" type="string">
  Auto-generated unique reference code in format: `{BRAND}{YEAR}{RANDOM}` (e.g., BMW2024A1B2)
</ResponseField>

<ResponseField name="brand" type="string">
  Vehicle brand name
</ResponseField>

<ResponseField name="model" type="string">
  Vehicle model name
</ResponseField>

<ResponseField name="hp" type="integer">
  Horsepower
</ResponseField>

<ResponseField name="autonomy" type="integer">
  Autonomy/range in kilometers
</ResponseField>

<ResponseField name="averageconsumption" type="double">
  Average fuel consumption
</ResponseField>

<ResponseField name="description" type="string">
  Vehicle description
</ResponseField>

<ResponseField name="price" type="decimal">
  Vehicle price
</ResponseField>

<ResponseField name="year" type="integer">
  Manufacturing year
</ResponseField>

<ResponseField name="fuelType" type="string">
  Fuel type (e.g., gasoline, diesel, electric, hybrid)
</ResponseField>

<ResponseField name="mileage" type="long">
  Mileage in kilometers
</ResponseField>

<ResponseField name="extras" type="string">
  Additional features and extras
</ResponseField>

<ResponseField name="createdAt" type="datetime">
  Timestamp when the vehicle was created
</ResponseField>

<ResponseField name="updatedAt" type="datetime">
  Timestamp when the vehicle was last updated
</ResponseField>

<ResponseField name="doors" type="integer">
  Number of doors
</ResponseField>

<ResponseField name="vehicleType" type="string">
  Type of vehicle (e.g., sedan, suv, truck, motorcycle)
</ResponseField>

<ResponseField name="offers" type="string">
  Special offers status (e.g., "Yes" or "No")
</ResponseField>

<ResponseField name="images" type="array">
  Array of vehicle images
  <ResponseField name="id" type="long">
    Image ID
  </ResponseField>
  <ResponseField name="imageUrl" type="string">
    Image URL
  </ResponseField>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "http://localhost:8080/api/vehicles/1"
```
</RequestExample>
